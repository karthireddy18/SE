pipeline {
    agent any

    tools {
        maven 'MAVEN-HOME'
    }

    stages {

        stage('Git Clone & Clean') {
            steps {
                bat 'rmdir /s /q mavenjava || exit 0'
                bat 'git clone https://github.com/Manisanthan/myjavaapp.git'
                bat 'mvn clean -f myjavaapp/pom.xml'
            }
        }

        stage('Install') {
            steps {
                bat 'mvn install -f myjavaapp/pom.xml'
            }
        }

        stage('Test') {
            steps {
                bat 'mvn test -f myjavaapp/pom.xml'
            }
        }

        stage('Package') {
            steps {
                bat 'mvn package -f myjavaapp/pom.xml'
            }
        }
    }
}
